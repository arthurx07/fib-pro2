64void Ciudad::comerciar(Ciudad &c)
65{
66 map<Producto, Atributos>::iterator it1 = inventario.begin();
67 map<Producto, Atributos>::iterator it2 = c.inventario.begin();
68 while (it1 != inventario.end() and it2 != c.inventario.end())
69 {
70 if (it1->first < it2->first) ++it1;
71 else if (it2->first < it1->first) ++it2;
72 else // if (it1->first == it2->first)
73 {
74 int dar = it1->second.posee - it1->second.necesita;
75 int rec = it2->second.necesita - it2->second.posee;
76 if (dar > 0 and rec > 0) rec = min(dar, rec);
77 else if (dar < 0 and rec < 0) rec = max(dar, rec);
78 else rec = 0;
79
80 it1->second.posee -= rec;
81 it2->second.posee += rec;
82
83 peso_total -= it1->first.consultar_peso()*rec;
84 volumen_total -= it1->first.consultar_volumen()*rec;
85 c.peso_total += it2->first.consultar_peso()*rec;
86 c.volumen_total += it2->first.consultar_volumen()*rec;
87
88 ++it1;
89 ++it2;
90 }
