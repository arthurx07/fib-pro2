153void Cuenca::buscar_ruta_recursiva(BinTree<string> t, stack<string> &ruta, int &comprar, int &vender) const
154{
155 if (not t.empty())
156 {
157 int idc = barco.consultar_id_comprar();
158 int idv = barco.consultar_id_vender();
159 if (ciudades.at(t.value()).tiene_producto(productos.at(idc)))
160 {
161 pair<int, int> atributos = ciudades.at(t.value()).consultar_producto(productos.at(barco.consultar_id_comprar()));
162 int diferencia = atributos.first - atributos.second; // posee - necesita
163 if (diferencia > 0) comprar -= min(diferencia, comprar);
164 }
165 if (ciudades.at(t.value()).tiene_producto(productos.at(idv)))
166 {
167 pair<int, int> atributos = ciudades.at(t.value()).consultar_producto(productos.at(barco.consultar_id_vender()));
168 int diferencia = atributos.second - atributos.first; // necesita - posee
169 if (diferencia > 0) vender -= min(diferencia, vender);
170 }
171
172 // nota: siempre comprar >= 0 y vender >=0
173 if (comprar > 0 or vender > 0)
174 {
175 int ci = comprar;
176 int vi = vender;
177 stack<string> ruta_izquierda;
178 buscar_ruta_recursiva(t.left(), ruta_izquierda, ci, vi);
179
180 int cd = comprar;
181 int vd = vender;
182 stack<string> ruta_derecha;
183 buscar_ruta_recursiva(t.right(), ruta_derecha, cd, vd);
184
185 if (not (comprar == ci and ci == cd and vender == vi and vi == vd))
186 {
187 if (ci + vi < cd + vd)
188 {
189 comprar = ci;
190 vender = vi;
191 ruta = ruta_izquierda;
192 }
193 else if (cd + vd < ci + vi)
194 {
195 comprar = cd;
196 vender = vd;
197 ruta = ruta_derecha;
198 }
199 else // if (ci + vi == cd + vd)
200 {
201 if (ruta_izquierda.size() <= ruta_derecha.size())
202 {
203 comprar = ci;
204 vender = vi;
205 ruta = ruta_izquierda;
206 }
207 else
208 {
209 comprar = cd;
210 vender = vd;
211 ruta = ruta_derecha;
212 }
213 }
214 }
215 }
216 ruta.push(t.value());

220int Cuenca::hacer_viaje()
221{
222 stack<string> ruta;
223
224 int cantc, cantv, comprar, vender;
225 cantc = comprar = barco.consultar_cantidad_comprar();
226 cantv = vender = barco.consultar_cantidad_vender();
227 buscar_ruta_recursiva(estructura, ruta, comprar, vender);
228
229 if (comprar != cantc or vender != cantv)
230 {
231 while (not ruta.empty())
232 {
233 string idc = ruta.top();
234 Producto pc = productos[barco.consultar_id_comprar()];
235 if (ciudades[idc].tiene_producto(pc))
236 {
237 pair<int, int> atributos = ciudades[idc].consultar_producto(pc);
238 int diferencia = atributos.first - atributos.second; // posee - necesita
239 if (diferencia > 0)
240 {
241 int minimo = min(diferencia, cantc);
242 ciudades[idc].modificar_producto(pc, atributos.first - minimo, atributos.second);
243 cantc -= minimo;
244 }
245 }
246 Producto pv = productos[barco.consultar_id_vender()];
247 if (ciudades[idc].tiene_producto(pv))
248 {
249 pair<int, int> atributos = ciudades[idc].consultar_producto(pv);
250 int diferencia = atributos.second - atributos.first; // necesita - posee
251 if (diferencia > 0)
252 {
253 int minimo = min(diferencia, cantv);
254 ciudades[idc].modificar_producto(pv, atributos.first + minimo, atributos.second);
255 cantv -= minimo;
256 }
257 }
258 ruta.pop();
259 if (ruta.empty()) barco.agregar_viaje(idc);
260 }
261 }
262
263 return barco.consultar_cantidad_comprar() + barco.consultar_cantidad_vender() - comprar - vender;
264}
